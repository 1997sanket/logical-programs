int check( char b[], int l2)
{
    int i;
    static int count=0;         //persists value
     
     
        
    
    for(i=0; i<(l2/2); i++)
    {
        if(b[i]==b[l2-1-i])
            continue;
        else
            break;
            
            
    
        
    }
    
    if((i==(l2/2)) && (i!=0))
     return ++count;            // if palindrome then return count 
    else 
        return 0;
    
}
